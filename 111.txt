            <!-- Номер заказа -->
            <TextBlock Text="Номер заказа" FontSize="14" Margin="0,0,0,6"/>
            <TextBox x:Name="txtOrderID" 
                     IsReadOnly="True" 
                     Height="40" 
                     FontSize="15" 
                     Background="#F8F8F8"/>

            <!-- Покупатель -->
            <TextBlock Text="Покупатель" FontSize="14" Margin="0,20,0,6"/>
            <ComboBox x:Name="cmbUser" 
                      Height="40" 
                      FontSize="15"/>

            <!-- Адрес доставки -->
            <TextBlock Text="Адрес доставки" FontSize="14" Margin="0,20,0,6"/>
            <TextBox x:Name="txtDeliveryAddress" 
                     Height="80" 
                     FontSize="15" 
                     TextWrapping="Wrap" 
                     AcceptsReturn="True" 
                     VerticalContentAlignment="Top"/>

            <!-- Дата заказа -->
            <TextBlock Text="Дата заказа" FontSize="14" Margin="0,20,0,6"/>
            <DatePicker x:Name="dpOrderDate" 
                        Height="40" 
                        FontSize="15"/>

            <!-- Дата доставки -->
            <TextBlock Text="Дата доставки" FontSize="14" Margin="0,20,0,6"/>
            <DatePicker x:Name="dpDeliveryDate"

            <!-- Статус заказа -->
            <TextBlock Text="Статус заказа" FontSize="14" Margin="0,20,0,6"/>
            <ComboBox x:Name="cmbStatus" 
                      Height="40" 
                      FontSize="15"/>

            <!-- Кнопки -->
            <StackPanel Orientation="Horizontal" Margin="0,40,0,0" HorizontalAlignment="Center">
                <Button x:Name="btnSave"

                <Button x:Name="btnCancel"

using System;
using System.Linq;
using System.Windows;
using System.Windows.Controls;
using Apotheke.Models;

namespace Apotheke.Pages
{
    public partial class AddEditOrderPage : Page
    {
        private Entities db = new Entities();
        private Orders _currentOrder = null;

        // Конструктор для добавления нового заказа
        public AddEditOrderPage()
        {
            InitializeComponent();
            LoadUsers();
            LoadStatuses();
            dpOrderDate.SelectedDate = DateTime.Now;
            txtOrderID.Text = "Новый заказ";
        }

        // Конструктор для редактирования существующего заказа
        public AddEditOrderPage(Orders order)
        {
            InitializeComponent();
            _currentOrder = order;
            LoadUsers();
            LoadStatuses();
            LoadOrderData();
        }

        private void LoadUsers()
        {
            cmbUser.ItemsSource = db.Users.OrderBy(u => u.FIO).ToList();
            cmbUser.DisplayMemberPath = "FIO";
            cmbUser.SelectedValuePath = "UserID";
        }

        private void LoadStatuses()
        {
            cmbStatus.ItemsSource = db.OrderStatus.ToList();
            cmbStatus.DisplayMemberPath = "StatusName";
            cmbStatus.SelectedValuePath = "StatusID";
        }

        private void LoadOrderData()
        {
            if (_currentOrder == null) return;

            txtOrderID.Text = $"Заказ №{_currentOrder.OrderID}";
            cmbUser.SelectedValue = _currentOrder.UserID;
            txtDeliveryAddress.Text = _currentOrder.DeliveryAddress;
            dpOrderDate.SelectedDate = _currentOrder.OrderDate;
            dpDeliveryDate.SelectedDate = _currentOrder.DeliveryDate;
            cmbStatus.SelectedValue = _currentOrder.StatusID;
        }

        private void btnSave_Click(object sender, RoutedEventArgs e)
        {
            if (cmbUser.SelectedValue == null)
            {
                MessageBox.Show("Выберите покупателя!", "Ошибка", MessageBoxButton.OK, MessageBoxImage.Warning);
                return;
            }

            if (string.IsNullOrWhiteSpace(txtDeliveryAddress.Text))
            {
                MessageBox.Show("Укажите адрес доставки!", "Ошибка", MessageBoxButton.OK, MessageBoxImage.Warning);
                return;
            }

            try
            {
                if (_currentOrder == null)
                {
                    // Добавление нового заказа
                    var newOrder = new Orders
                    {
                        UserID = (int)cmbUser.SelectedValue,
                        DeliveryAddress = txtDeliveryAddress.Text.Trim(),
                        OrderDate = dpOrderDate.SelectedDate ?? DateTime.Now,
                        DeliveryDate = dpDeliveryDate.SelectedDate,
                        StatusID = cmbStatus.SelectedValue as int?
                    };

                    db.Orders.Add(newOrder);
                    db.SaveChanges();

                    MessageBox.Show("Новый заказ успешно создан!", "Успех", MessageBoxButton.OK, MessageBoxImage.Information);
                }
                else
                {
                    // Редактирование заказа
                    _currentOrder.UserID = (int)cmbUser.SelectedValue;
                    _currentOrder.DeliveryAddress = txtDeliveryAddress.Text.Trim();
                    _currentOrder.OrderDate = dpOrderDate.SelectedDate ?? DateTime.Now;
                    _currentOrder.DeliveryDate = dpDeliveryDate.SelectedDate;
                    _currentOrder.StatusID = cmbStatus.SelectedValue as int?;

                    db.SaveChanges();
                    MessageBox.Show("Заказ успешно обновлён!", "Успех", MessageBoxButton.OK, MessageBoxImage.Information);
                }

                // Возвращаемся на список заказов
                MainWindow.Instance.NavigateTo(new OrdersPage());
            }
            catch (Exception ex)
            {
                MessageBox.Show($"Ошибка при сохранении заказа:\n{ex.Message}", "Ошибка", MessageBoxButton.OK, MessageBoxImage.Error);
            }
        }

        private void btnCancel_Click(object sender, RoutedEventArgs e)
        {
            MainWindow.Instance.NavigateTo(new OrdersPage());
        }
    }
}
