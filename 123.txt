<Page x:Class="Apotheke.Pages.OrdersPage"
      xmlns="http://schemas.microsoft.com/winfx/2006/xaml/presentation"
      xmlns:x="http://schemas.microsoft.com/winfx/2006/xaml"
      xmlns:mc="http://schemas.openxmlformats.org/markup-compatibility/2006"
      xmlns:d="http://schemas.microsoft.com/expression/blend/2008"
      mc:Ignorable="d"
      Title="OrdersPage"
      Background="#F4F6F9">

    <Grid Margin="20">
        <Grid.RowDefinitions>
            <RowDefinition Height="Auto"/>
            <RowDefinition Height="*"/>
        </Grid.RowDefinitions>

        <!-- Заголовок и кнопки -->
        <Grid Margin="0,0,0,20">
            <TextBlock Text="Заказы" 
                       FontSize="32" 
                       FontWeight="SemiBold" 
                       Foreground="#2C3E50"
                       HorizontalAlignment="Left"/>

            <StackPanel Orientation="Horizontal" HorizontalAlignment="Right">
                <Button x:Name="btnAddOrder" 
                        Content="Добавить заказ" 
                        Width="170" 
                        Height="42"
                        Background="#27AE60" 
                        Foreground="White"
                        Margin="10,0"
                        Click="btnAddOrder_Click"
                        Visibility="Collapsed"/>

                <Button x:Name="btnDeleteOrder" 
                        Content="Удалить выбранный" 
                        Width="170" 
                        Height="42"
                        Background="#E74C3C" 
                        Foreground="White"
                        Margin="10,0"
                        Click="btnDeleteOrder_Click"
                        Visibility="Collapsed"/>

                <Button x:Name="btnRefresh" 
                        Content="Обновить" 
                        Width="120" 
                        Height="42"
                        Background="#3498DB" 
                        Foreground="White"
                        Click="btnRefresh_Click"/>
            </StackPanel>
        </Grid>

        <!-- Список заказов -->
        <ScrollViewer Grid.Row="1" VerticalScrollBarVisibility="Auto">
            <StackPanel x:Name="stackOrders" />
        </ScrollViewer>
    </Grid>
</Page>


using System;
using System.Linq;
using System.Windows;
using System.Windows.Controls;
using System.Windows.Input;
using System.Windows.Media;
using Apotheke.Models;
using System.Data.Entity;

namespace Apotheke.Pages
{
    public partial class OrdersPage : Page
    {
        private Entities db = new Entities();
        private Orders _selectedOrder = null;

        public OrdersPage()
        {
            InitializeComponent();
            LoadOrders();
            UpdateButtonsVisibility();
        }

        private void LoadOrders()
        {
            stackOrders.Children.Clear();
            _selectedOrder = null;

            var orders = db.Orders
                           .Include(o => o.Users)
                           .Include(o => o.OrderStatus)
                           .Include(o => o.OrderItems.Select(oi => oi.Products))
                           .OrderByDescending(o => o.OrderDate)
                           .ToList();

            foreach (var order in orders)
            {
                var card = CreateOrderCard(order);
                stackOrders.Children.Add(card);
            }
        }

        private Border CreateOrderCard(Orders order)
        {
            var border = new Border
            {
                Background = Brushes.White,
                CornerRadius = new CornerRadius(10),
                Margin = new Thickness(0, 8, 0, 8),
                Padding = new Thickness(20),
                BorderBrush = new SolidColorBrush(Color.FromRgb(220, 220, 220)),
                BorderThickness = new Thickness(1),
                Cursor = Cursors.Hand
            };

            var grid = new Grid();
            grid.ColumnDefinitions.Add(new ColumnDefinition { Width = new GridLength(1, GridUnitType.Star) });
            grid.ColumnDefinitions.Add(new ColumnDefinition { Width = new GridLength(200) });

            // Левая часть
            var leftStack = new StackPanel();

            var title = new TextBlock
            {
                Text = $"Заказ №{order.OrderID}",
                FontSize = 18,
                FontWeight = FontWeights.SemiBold,
                Foreground = new SolidColorBrush(Color.FromRgb(44, 62, 80))
            };

            var customer = new TextBlock
            {
                Text = $"Покупатель: {order.Users?.FIO ?? "—"}",
                FontSize = 14,
                Margin = new Thickness(0, 8, 0, 0)
            };

            var status = new TextBlock
            {
                Text = $"Статус: {order.OrderStatus?.StatusName ?? "Не указан"}",
                FontSize = 14,
                Margin = new Thickness(0, 4, 0, 0)
            };

            var address = new TextBlock
            {
                Text = $"Адрес: {order.DeliveryAddress ?? "—"}",
                FontSize = 14,
                TextWrapping = TextWrapping.Wrap,
                Margin = new Thickness(0, 4, 0, 0)
            };

            var orderDate = new TextBlock
            {
                Text = $"Дата заказа: {order.OrderDate?.ToString("dd.MM.yyyy HH:mm") ?? "—"}",
                FontSize = 14,
                Margin = new Thickness(0, 4, 0, 0)
            };

            leftStack.Children.Add(title);
            leftStack.Children.Add(customer);
            leftStack.Children.Add(status);
            leftStack.Children.Add(address);
            leftStack.Children.Add(orderDate);

            // Правая часть - Дата доставки
            var rightStack = new StackPanel { HorizontalAlignment = HorizontalAlignment.Right, VerticalAlignment = VerticalAlignment.Center };

            var delLabel = new TextBlock
            {
                Text = "Дата доставки",
                FontSize = 13,
                Foreground = Brushes.Gray,
                HorizontalAlignment = HorizontalAlignment.Right
            };

            var delDate = new TextBlock
            {
                Text = order.DeliveryDate?.ToString("dd.MM.yyyy") ?? "—",
                FontSize = 18,
                FontWeight = FontWeights.SemiBold,
                Foreground = new SolidColorBrush(Color.FromRgb(39, 174, 96)),
                HorizontalAlignment = HorizontalAlignment.Right,
                Margin = new Thickness(5, 0, 0, 0)
            };

            rightStack.Children.Add(delLabel);
            rightStack.Children.Add(delDate);

            grid.Children.Add(leftStack);
            grid.Children.Add(rightStack);
            Grid.SetColumn(leftStack, 0);
            Grid.SetColumn(rightStack, 1);

            border.Child = grid;

            // Двойной клик — просмотр состава заказа
            border.MouseLeftButtonDown += (s, e) =>
            {
                if (e.ClickCount == 2)
                    ShowOrderDetails(order);
            };

            // Одинарный клик — выбор для удаления
            border.MouseLeftButtonUp += (s, e) =>
            {
                _selectedOrder = order;
                HighlightSelectedCard(border);
            };

            return border;
        }

        private void HighlightSelectedCard(Border selectedBorder)
        {
            foreach (Border card in stackOrders.Children.OfType<Border>())
            {
                card.BorderBrush = new SolidColorBrush(Color.FromRgb(220, 220, 220));
                card.BorderThickness = new Thickness(1);
            }

            selectedBorder.BorderBrush = new SolidColorBrush(Color.FromRgb(231, 76, 60));
            selectedBorder.BorderThickness = new Thickness(3);
        }

        private void ShowOrderDetails(Orders order)
        {
            string text = $"Заказ №{order.OrderID}\n\n" +
                          $"Покупатель: {order.Users?.FIO ?? "—"}\n" +
                          $"Дата заказа: {order.OrderDate?.ToString("dd.MM.yyyy HH:mm")}\n" +
                          $"Адрес: {order.DeliveryAddress ?? "—"}\n\n" +
                          "Состав заказа:\n";

            if (order.OrderItems?.Any() == true)
            {
                foreach (var item in order.OrderItems)
                {
                    text += $"• {item.Products?.ProductName ?? "—"} — {item.Quantity} шт. × {item.PriceInOrder} ₽ = {item.LineTotal} ₽\n";
                }
            }
            else
            {
                text += "— Товары отсутствуют —";
            }

            MessageBox.Show(text, "Подробности заказа", MessageBoxButton.OK, MessageBoxImage.Information);
        }

        // ====================== Кнопки ======================

        private void btnAddOrder_Click(object sender, RoutedEventArgs e)
        {
            MainWindow.Instance.NavigateTo(new AddEditOrderPage());
        }

        private void btnDeleteOrder_Click(object sender, RoutedEventArgs e)
        {
            if (_selectedOrder == null)
            {
                MessageBox.Show("Сначала выберите заказ!", "Внимание", MessageBoxButton.OK, MessageBoxImage.Warning);
                return;
            }

            var result = MessageBox.Show($"Удалить заказ №{_selectedOrder.OrderID}?\n\n" +
                                       $"Покупатель: {_selectedOrder.Users?.FIO}",
                                       "Подтверждение удаления",
                                       MessageBoxButton.YesNo, MessageBoxImage.Warning);

            if (result == MessageBoxResult.Yes)
            {
                try
                {
                    db.Orders.Remove(_selectedOrder);
                    db.SaveChanges();

                    MessageBox.Show("Заказ успешно удалён!", "Успех");
                    LoadOrders();
                }
                catch (Exception ex)
                {
                    MessageBox.Show("Ошибка удаления: " + ex.Message, "Ошибка", MessageBoxButton.OK, MessageBoxImage.Error);
                }
            }
        }

        private void btnRefresh_Click(object sender, RoutedEventArgs e)
        {
            LoadOrders();
        }

        private void UpdateButtonsVisibility()
        {
            bool isAdmin = CurrentSession.IsAdmin;

            btnAddOrder.Visibility = isAdmin ? Visibility.Visible : Visibility.Collapsed;
            btnDeleteOrder.Visibility = isAdmin ? Visibility.Visible : Visibility.Collapsed;
        }
    }
}
